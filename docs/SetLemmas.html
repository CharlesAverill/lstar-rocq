Library SetLemmas

From Stdlib Require Import List.

Section SL.

Variable X : Type.
Variable eq_dec : ∀ (x y : X), {x = y} + {x ≠ y}.

Definition finite (f : X → bool) :=
    {l : list X | NoDup l ∧
        ∀ (s : X), f s = true ↔ In s l}.

Definition num_states_of_fin {f : X → bool} (H : finite f) : nat :=
    match H with
    | exist _ x _ ⇒ List.length x
    end.

Updating sets
Definition update (S : X → bool) k b :=
    fun s ⇒ if eq_dec s k then b else S s.

Notation "s [ k := v ]" := (update s k v).

Lemma update_neq : ∀ S x y k,
    x ≠ y →
    S[x := k] y = S y.

Lemma update_eq : ∀ S x k,
    S[x := k] x = k.

Lemma finite_subset_is_smaller : ∀
    (f g : X → bool)
    (FinF : finite f)
    (FinG : finite g)
    (FsubG : ∀ (x : X), f x = true → g x = true),
    num_states_of_fin FinF ≤ num_states_of_fin FinG.

Lemma finite_update_impl_finite : ∀
    (f : X → bool) k v
    (FinUpdF : finite f[k := v]),
    finite f.

End SL.
